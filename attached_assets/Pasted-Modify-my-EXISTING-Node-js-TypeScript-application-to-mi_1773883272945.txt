import { QUESTION_BANK } from "./questionBank";

export async function runQuestionAudit(reportText: string) {

  const questionsText = QUESTION_BANK.map(q => `${q.id}: ${q.text}`).join("\n");

  const prompt = USER_PROMPT_TEMPLATE
    .replace("{{QUESTIONS}}", questionsText)
    .replace("{{REPORT}}", reportText);

  const response = await openai.chat.completions.create({
    model: "gpt-5",
    temperature: 0,
    messages: [
      { role: "system", content: SYSTEM_PROMPT },
      { role: "user", content: prompt }
    ]
  });

  return JSON.parse(response.choices[0].message.content);
}

function scoreAnswer(answer: string, weight: number) {
  switch (answer) {
    case "PASS": return weight;
    case "PARTIAL": return weight * 0.5;
    case "FAIL": return 0;
    case "NOT_APPLICABLE": return weight;
    default: return 0;
  }
}

export function computeScore(results, questionBank) {

  let total = 0;
  let max = 0;

  const sectionTotals = {};

  results.forEach(r => {
    const q = questionBank.find(q => q.id === r.id);
    if (!q) return;

    const score = scoreAnswer(r.answer, q.weight);

    total += score;
    max += q.weight;

    if (!sectionTotals[q.section]) {
      sectionTotals[q.section] = { score: 0, max: 0 };
    }

    sectionTotals[q.section].score += score;
    sectionTotals[q.section].max += q.weight;
  });

  return {
    total,
    max,
    percent: Math.round((total / max) * 100),
    sections: sectionTotals
  };
}
