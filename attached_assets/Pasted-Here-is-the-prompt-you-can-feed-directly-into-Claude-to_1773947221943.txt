```typescript
export function validateStackOrder(documentPages: string[]): boolean {
  // Expected flow based on carrier SLAs
  const expectedOrder = [
    "Desk Adjuster Report",
    "STATEMENT OF LOSS",
    "Dear", // Payment Letter
    "Estimate",
    "Photo Sheet",
    "Sketch", // or Diagram
    "ISO File Number" // Prior Loss Report
  ];

  let currentIndex = 0;
  
  for (const page of documentPages) {
    if (currentIndex >= expectedOrder.length) break;
    
    // If the current page contains the next expected document header, advance the index
    if (page.includes(expectedOrder[currentIndex])) {
      currentIndex++;
    }
  }

  // Returns true if all documents were found in the correct sequential order
  return currentIndex === expectedOrder.length;
}
```

```typescript
  });
}
```
