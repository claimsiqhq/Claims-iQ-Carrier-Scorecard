export function runValidation(reportText: string) {
  const issues = {
    critical: [],
    warnings: [],
    info: []
  };

  const payments = reportText.match(/\$[\d,]+/g) || [];
  if (new Set(payments).size > 3) {
    issues.warnings.push({
      type: "payment_variance",
      message: "Multiple payment values detected"
    });
  }

  if (/deferred|pending/i.test(reportText)) {
    if (!/next step|follow up/i.test(reportText)) {
      issues.warnings.push({
        type: "deferred_missing_next_step",
        message: "Deferred item missing next step"
      });
    }
  }

  const ready = issues.critical.length === 0;

  return { ...issues, ready };
}

import { QUESTION_BANK } from "./questionBank";
import { SYSTEM_PROMPT, USER_PROMPT_TEMPLATE } from "./prompts";

export async function runQuestionAudit(reportText: string) {

  const questionsText = QUESTION_BANK
    .map(q => `${q.id}: ${q.text}`)
    .join("\\n");

  const userPrompt = USER_PROMPT_TEMPLATE
    .replace("{{QUESTIONS}}", questionsText)
    .replace("{{REPORT}}", reportText);

  const response = await openai.chat.completions.create({
    model: "gpt-5",
    temperature: 0,
    messages: [
      { role: "system", content: SYSTEM_PROMPT },
      { role: "user", content: userPrompt }
    ]
  });

  return JSON.parse(response.choices[0].message.content);
}

function scoreAnswer(answer, weight) {
  if (answer === "PASS") return weight;
  if (answer === "PARTIAL") return weight * 0.5;
  if (answer === "FAIL") return 0;
  if (answer === "NOT_APPLICABLE") return weight;
  return 0;
}

export function computeScore(results, questionBank) {
  let total = 0;
  let max = 0;

  results.forEach(r => {
    const q = questionBank.find(q => q.id === r.id);
    if (!q) return;

    total += scoreAnswer(r.answer, q.weight);
    max += q.weight;
  });

  return {
    total,
    max,
    percent: Math.round((total / max) * 100)
  };
}
