export function groupByRootIssue(results) {
  const groups = {};

  results.forEach(r => {
    const key = r.root_issue || r.id;

    if (!groups[key]) {
      groups[key] = [];
    }

    groups[key].push(r);
  });

  return groups;
}

function scoreAnswer(answer, weight) {
  if (answer === "PASS") return weight;
  if (answer === "PARTIAL") return weight * 0.5;
  if (answer === "FAIL") return 0;
  if (answer === "NOT_APPLICABLE") return weight;
  return 0;
}

export function computeScore(results, questionBank) {

  const groups = groupByRootIssue(results);

  let total = 0;
  let max = 0;

  Object.values(groups).forEach(group => {

    const primary = group.find(q => q.answer !== "PASS");

    if (!primary) {
      // all PASS
      group.forEach(q => {
        const qb = questionBank.find(x => x.id === q.id);
        total += qb.weight;
        max += qb.weight;
      });
      return;
    }

    group.forEach((q, i) => {
      const qb = questionBank.find(x => x.id === q.id);
      max += qb.weight;

      if (i === 0) {
        total += scoreAnswer(q.answer, qb.weight);
      } else {
        total += scoreAnswer(q.answer, qb.weight) * 0.25;
      }
    });

  });

  return {
    total,
    max,
    percent: Math.round((total / max) * 100)
  };
}
